#一维高斯概率密度函数
def OneDimentionGB(X,u,s):
    ALLF=[];
    for IX in range(0,len(X)):
        f=math.exp(-math.pow((X[IX]-u),2)/2*s)/math.sqrt(2*math.pi*s); ALLF.append(f);
    return ALLF;
